/-
AbcExtraction.lean – Algorithme d'extraction (on attend `none`).
-/

import .AbcPillars
import Kernel.DRSP

open SpectralLibrary

variable (N0 : ℕ) (hN0 : N0 > 0)

def abc_solver : Option (ℕ × ℕ × ℕ) :=
  let H := H_abc N0 hN0
  match drsp_solver H with
  | none => none
  | some σ => some (decode_triple σ)

/-- Axiome : l'instance SAT est insatisfiable (pas de contre‑exemple). -/
axiom abc_unsat : ¬ Satisfiable (abc_SAT N0)

theorem extraction_unsat : abc_solver N0 hN0 = none :=
  by
    have h_unsat := abc_unsat N0
    have h_correct := drsp_correct_abc N0 hN0
    rcases h_correct with ⟨alg, h_poly, h_equiv⟩
    contrapose h_unsat
    intro h_ne
    have h_sat : Satisfiable (abc_SAT N0) := by
      obtain ⟨σ, hσ⟩ := Option.ne_none_iff_exists.1 h_ne
      exact ⟨σ, hσ⟩
    exact h_unsat
