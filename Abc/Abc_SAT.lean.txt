/-
AbcSAT.lean – Réduction à 3‑SAT via Tseitin.
-/

import .AbcCircuit
import Kernel.Tseitin

open Circuit

def abc_SAT (N0 : ℕ) : SATInstance :=
  tseitin (abc_circuit N0)

theorem abc_SAT_correct (N0 : ℕ) :
    Satisfiable (abc_SAT N0) ↔ ∃ a b c : ℕ, a ≤ N0 ∧ b ≤ N0 ∧ c ≤ N0 ∧
      a + b = c ∧ gcd a (gcd b c) = 1 ∧ c > C_eps * (rad (a*b*c))^2 :=
  by
    rw [tseitin_correct (abc_circuit N0)]
    exact abc_circuit_correct N0
