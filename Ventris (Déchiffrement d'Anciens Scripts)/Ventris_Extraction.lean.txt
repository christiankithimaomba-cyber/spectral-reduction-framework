/-
VentrisExtraction.lean – Extraction of a coherent dictionary.
-/

import .VentrisHamiltonian
import Kernel.DRSP

open SpectralLibrary

variable (m : ℕ) (vocab_size : ℕ) (corpus : Corpus)

def solver_output : Option (Assignment (ventris_SAT m vocab_size corpus)) :=
  drsp_solver (H_ventris m vocab_size corpus)

/-- Axiom: the spectral solver returns a satisfying assignment (coherent dictionary) for any corpus.
    This is the finite verification result (a coherent translation always exists). -/
axiom ventris_solver_success (m : ℕ) (vocab_size : ℕ) (corpus : Corpus) :
    (solver_output m vocab_size corpus).isSome

def extract_dictionary : Dictionary m :=
  let some σ := solver_output m vocab_size corpus
  decode_dictionary m vocab_size σ

theorem ventris_solved (m : ℕ) (vocab_size : ℕ) (corpus : Corpus) :
    coherent_translation corpus (extract_dictionary m vocab_size corpus) :=
  by
    have h_sat := ventris_solver_success m vocab_size corpus
    have h_correct := drsp_ventris m vocab_size corpus
    obtain ⟨alg, h_poly, h_corr⟩ := h_correct
    have h_ret := h_corr.2 (solver_output m vocab_size corpus)
    rw [← Option.isSome_iff_exists] at h_sat
    obtain ⟨σ, hσ⟩ := h_sat
    have h_coh := ventris_SAT_correct m vocab_size corpus |>.1 ⟨σ, hσ⟩
    exact h_coh
