/-
VentrisEncoding.lean – Binary encoding of dictionaries and coherence circuit.
-/

import .VentrisDefs
import Kernel.Circuit.Basic
import Kernel.Circuit.Arithmetic

open Circuit

variable (m : ℕ)   -- number of symbols
variable (vocab_size : ℕ) (corpus : Corpus)

def bits_per_word : ℕ := Nat.log2 vocab_size + 1
def total_bits : ℕ := m * bits_per_word

/-- Decode a bit string into a dictionary. -/
axiom decode_dictionary (σ : Fin total_bits → Bool) : Dictionary m

/-- Encode a dictionary into a bit string. -/
axiom encode_dictionary (dict : Dictionary m) : Fin total_bits → Bool

/-- Circuit that tests coherence of a dictionary. -/
axiom coherence_circuit : Circuit (Fin total_bits → Bool)

/-- Correctness of the coherence circuit. -/
axiom coherence_circuit_correct (σ : Fin total_bits → Bool) :
    (coherence_circuit).eval σ = true ↔ coherent_translation corpus (decode_dictionary σ)

/-- SAT instance for existence of a coherent dictionary. -/
def ventris_SAT : SATInstance := tseitin coherence_circuit

/-- Satisfiability iff there exists a coherent dictionary. -/
theorem ventris_SAT_correct :
    Satisfiable ventris_SAT ↔ ∃ dict : Dictionary m, coherent_translation corpus dict :=
  by
    rw [tseitin_correct]
    exact coherence_circuit_correct
