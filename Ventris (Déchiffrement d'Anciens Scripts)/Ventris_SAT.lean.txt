/-
VentrisSAT.lean – SAT instance for coherent dictionary.
-/

import .VentrisGrammar
import Kernel.Tseitin

open Circuit

variable (m : ℕ) (vocab_size : ℕ) (corpus : Corpus)

def ventris_SAT : SATInstance :=
  tseitin (grammar_circuit m vocab_size corpus)

theorem ventris_SAT_correct :
    Satisfiable (ventris_SAT m vocab_size corpus) ↔
    ∃ dict : Dictionary m, coherent_translation corpus dict :=
  by
    rw [tseitin_correct]
    exact grammar_circuit_correct m vocab_size corpus
