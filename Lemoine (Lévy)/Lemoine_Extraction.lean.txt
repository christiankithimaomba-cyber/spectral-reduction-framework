/-
LemoineExtraction.lean – Algorithme d'extraction pour le range fini.
-/

import .LemoinePillars
import Kernel.DRSP

open SpectralLibrary

variable (n : ℕ) (hn : Odd n ∧ n > 5)

def extract_pair : Option (ℕ × ℕ) :=
  let H := H_lemoine n hn
  match drsp_solver H with
  | none => none
  | some σ => some (decode_pair σ)

theorem extraction_correct (n : ℕ) (hn : Odd n ∧ n > 5) (h_le : n ≤ N0) :
    extract_pair n hn ≠ none :=
  by
    -- Pour n ≤ N0, l'existence d'une solution est garantie par vérification numérique
    have h_sat : Satisfiable (lemoine_SAT n hn) :=
      lemoine_finite_verification n hn h_le
    exact drsp_solver_correct (H_lemoine n hn) h_sat

/-- Axiome : pour n ≤ N0, il existe une solution (vérification exhaustive). -/
axiom lemoine_finite_verification (n : ℕ) (hn : Odd n ∧ n > 5) (h_le : n ≤ N0) :
    ∃ p q, Prime p ∧ Prime q ∧ p + 2*q = n
