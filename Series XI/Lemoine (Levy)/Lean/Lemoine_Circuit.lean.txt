/-
LemoineCircuit.lean – Circuit booléen pour tester p + 2q = n avec p,q premiers.
-/

import .LemoineDefs
import Kernel.Circuit.Basic
import Kernel.Circuit.Arithmetic
import Kernel.Circuit.Prime

open Circuit

variable (n : ℕ) (hn : Odd n ∧ n > 5)

def L : ℕ := Nat.log2 n + 1

/-- Circuit pour le test de primalité. -/
def prime_circuit_impl (x : Circuit (List Bool)) : Circuit Bool :=
  prime_circuit x   -- du kernel, correction axiome

/-- Circuit principal : vérifie p + 2q = n et primalité de p et q. -/
def lemoine_circuit : Circuit (Fin (2*L) → Bool) :=
  let p := input 0 L
  let q := input L L
  let two_q := shift_left q 1   -- 2q
  let s := add p two_q
  let eq := eq s (constant n)
  let pr_p := prime_circuit_impl p
  let pr_q := prime_circuit_impl q
  and (and eq pr_p) pr_q

/-- Axiome : correction du circuit (preuve standard, cf. kernel). -/
axiom lemoine_circuit_correct (bits : Fin (2*L) → Bool) :
    (lemoine_circuit n hn).eval bits = true ↔
    let p := bits_to_nat (bits.slice 0 L)
    let q := bits_to_nat (bits.slice L L)
    Prime p ∧ Prime q ∧ p + 2*q = n
