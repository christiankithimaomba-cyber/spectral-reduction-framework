import Kernel.MPS
import PNP.Compliance

open MPS

/-- Axiome de Brandão‑Horodecki. -/
axiom brandao_horodecki {V : Type*} [Fintype V] [DecidableEq V] {P : SpectralProblem V}
    (h : BrandaoHorodeckiAssumptions P) :
    ∃ C > 0, ∀ B : Finset V, entanglement_entropy (ground_state P) B ≤ C * Real.log (Fintype.card V)

/-- Loi d'aire pour un problème satisfaisant les hypothèses. -/
theorem area_law {V : Type*} [Fintype V] [DecidableEq V] {P : SpectralProblem V}
    (h : BrandaoHorodeckiAssumptions P) :
    ∃ C > 0, ∀ B : Finset V, entanglement_entropy (ground_state P) B ≤ C * Real.log (Fintype.card V) :=
  brandao_horodecki h
