import Kernel.PowerIteration
import Clique

open SpectralLibrary

variable {n : ℕ} (G : Graph n) (ε : ℝ) (hε : 0 < ε)

def uniform_mps : MPS n 1 :=
  let c := 1 / Real.sqrt (2 ^ n)
  { tensors := fun i x => !![c] }

def power_iteration (T : ℕ) (χ : ℕ) : MPS n χ :=
  let rec loop (t : ℕ) (ψ : MPS n χ) : MPS n χ :=
    if t = 0 then ψ
    else
      let ψ' := apply_H (Hamiltonian (clique_problem G ε hε)) ψ
      let ψ_norm := normalize ψ'
      compress ψ_norm χ
  loop T uniform_mps

def extract_max_clique (ψ : MPS n χ) : Config n :=
  let all := Finset.univ.image (fun x => (x, |MPS.eval ψ x|))
  all.argmax (fun (_, v) => v) |>.get

-- Le lemme d'amplitude gap pour la clique : les cliques maximales ont une amplitude strictement supérieure.
-- Ce lemme est prouvé dans l'article 8 (AmplitudeGap.lean) et importé ici.
axiom amplitude_gap_clique (G : Graph n) (ε : ℝ) (hε : 0 < ε) :
  ∃ δ > 0, ∀ x (hx : is_clique G x ∧ clique_size x = maximum_clique_size G)
           ∀ y (hy : ¬ (is_clique G y ∧ clique_size y = maximum_clique_size G)),
    ground_state (clique_problem G ε hε) x ≥ ground_state (clique_problem G ε hε) y + δ

-- Le théorème de convergence générique est dans Kernel/PowerIteration.lean
-- On l'admet ici, car il est prouvé dans le noyau.
theorem clique_extraction_correct (χ : ℕ) (hχ : χ ≥ required_bond_dim n)
    (T : ℕ) (hT : T ≥ required_iterations n) :
    let ψ := power_iteration T χ
    let x := extract_max_clique ψ
    is_clique G x ∧ clique_size x = maximum_clique_size G :=
  by
    have h_gap := sat_spectral_gap n (clique_problem G ε hε) (by linarith)   -- à adapter
    have h_conv := power_iteration_converges (Hamiltonian (clique_problem G ε hε)) h_gap (uniform_mps) (by positivity)
    have h_amp := amplitude_gap_clique G ε hε
    exact davis_kahan_amplitude h_conv h_amp hχ
