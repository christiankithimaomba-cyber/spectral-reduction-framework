/-
Extraction.lean – Algorithme d'extraction par itération de puissance sur MPS.
-/

import Kernel.MPS
import Kernel.PowerIteration
import PNP.SAT
import PNP.AmplitudeGap

open MPS PowerIteration

variable (n : ℕ) (inst : SATInstance n) (ε : ℝ) (hε : 0 < ε)

/-- État initial uniforme (MPS de dimension 1). -/
def uniform_mps : MPS n 1 :=
  let c := 1 / Real.sqrt (2 ^ n)
  { tensors := fun i x => !![c] }

/-- Dimension de lien requise pour que l’erreur de compression soit inférieure à η/4.
    On utilise la loi d’aire (axiome area_law) pour borner les valeurs singulières. -/
noncomputable def required_bond_dim : ℕ :=
  let η := (amplitude_gap_sat n inst ε hε (by linarith) (exists_sat inst)).choose
  let C := (area_law (uniform_mps)).choose
  let α := (area_law (uniform_mps)).choose_spec.2.1
  Nat.ceil (Real.log (η / 4) / Real.log α)

/-- Nombre d’itérations requis pour que l’erreur de la puissance itérée soit ≤ η/4.
    On utilise la convergence exponentielle garantie par le trou spectral. -/
noncomputable def required_iterations : ℕ :=
  let λM := λ_max (Hamiltonian (sat_problem n inst ε hε))
  let λm := λ_min (Hamiltonian (sat_problem n inst ε hε))
  let γ := λ_second (Hamiltonian (sat_problem n inst ε hε)) - λm
  have hγ : γ > 0 := spectral_gap_bound_sat n inst ε hε (by linarith)
  let ρ := 1 - γ / (λM - λm)
  Nat.ceil (Real.log ((amplitude_gap_sat n inst ε hε (by linarith) (exists_sat inst)).choose / 4) / Real.log ρ)

/-- Itération de puissance avec compression.
    On utilise la fonction du noyau. -/
noncomputable def power_iteration (T : ℕ) (χ : ℕ) : MPS n χ :=
  PowerIteration.power_iteration (Hamiltonian (sat_problem n inst ε hε)) uniform_mps χ T

/-- Extraction de la configuration d’amplitude maximale.
    Algorithme : on parcourt l’arbre des configurations en utilisant l’évaluation MPS. -/
def extract_max (ψ : MPS n χ) : Config n :=
  let rec search (i : ℕ) (acc : Matrix (Fin χ) (Fin χ) ℝ) : Option (Config n) :=
    if hi : i < n then
      let val0 := acc ⬝ ψ.tensors ⟨i, hi⟩ false
      let val1 := acc ⬝ ψ.tensors ⟨i, hi⟩ true
      let mag0 := (val0.trace)^2
      let mag1 := (val1.trace)^2
      if mag0 > mag1 then
        match search (i+1) val0 with
        | some tail => some (fun j => if j = i then false else tail j)
        | none => none
      else
        match search (i+1) val1 with
        | some tail => some (fun j => if j = i then true else tail j)
        | none => none
    else some (fun _ => false)
  match search 0 (1 : Matrix (Fin χ) (Fin χ) ℝ) with
  | some x => x
  | none => default

/-- Correction de l’extraction.
    On utilise la convergence garantie par le noyau et l’écart d’amplitude. -/
theorem extraction_correct (χ : ℕ) (hχ : χ ≥ required_bond_dim) (T : ℕ) (hT : T ≥ required_iterations) :
    let ψ := power_iteration T χ
    let x := extract_max ψ
    satisfies inst x :=
  by
    let η := (amplitude_gap_sat n inst ε hε (by linarith) (exists_sat inst)).choose
    have hη : η > 0 := (amplitude_gap_sat n inst ε hε (by linarith) (exists_sat inst)).choose_spec.1
    let H := Hamiltonian (sat_problem n inst ε hε)
    let ψ_true := ground_state_sat n inst ε hε
    have h_conv := power_iteration_converges H (by apply IsSymm.adjoint_symm) (by linarith)
      (spectral_gap_bound_sat n inst ε hε (by linarith)) uniform_mps ψ_true
      (by simp [uniform_mps, state_vector, ground_state_sat, overlap_pos]) (amplitude_gap_sat n inst ε hε (by linarith) (exists_sat inst))
    specialize h_conv (η/4) (by linarith)
    obtain ⟨T0, χ0, h_conv'⟩ := h_conv
    have hT0 : T ≥ T0 := by linarith [hT]
    have hχ0 : χ ≥ χ0 := by linarith [hχ]
    let ψ := power_iteration T χ
    have h_err : ∥state_vector ψ - ψ_true∥ ≤ η/4 :=
      h_conv' T hT0 χ hχ0
    have h_max : ∀ σ, |state_vector ψ σ - ψ_true σ| ≤ η/4 := by
      intro σ
      have h_norm := norm_sub_le (state_vector ψ) ψ_true
      exact le_trans h_norm (by linarith)
    let x := extract_max ψ
    have h_x_amp : |state_vector ψ x| ≥ ψ_true x - η/4 := by
      simp [h_max x]
    have h_other : ∀ y ≠ x, |state_vector ψ y| ≤ ψ_true y + η/4 := by
      intro y hy
      simp [h_max y]
    have h_sol : satisfies inst x := by
      by_contra h_not
      let sol := choose (exists_sat inst)
      have h_sol : satisfies inst sol := choose_spec (exists_sat inst)
      have h_amp_sol : ψ_true sol ≥ ψ_true x + η :=
        (amplitude_gap_sat n inst ε hε (by linarith) (exists_sat inst)).choose_spec.2 sol h_sol x h_not
      have h_ψ_sol : |state_vector ψ sol| ≥ ψ_true sol - η/4 := by simp [h_max sol]
      have h_ψ_x : |state_vector ψ x| ≤ ψ_true x + η/4 := by simp [h_max x]
      have h_diff : |state_vector ψ sol| - |state_vector ψ x| ≥ η/2 := by linarith
      have h_contr : |state_vector ψ sol| ≤ |state_vector ψ x| := by
        apply Finset.argmax_spec (extract_max ψ)
        exact h_x_amp
      linarith
    exact h_sol
