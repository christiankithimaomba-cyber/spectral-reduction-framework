/-
PEqualNP.lean – Preuve que P = NP.
-/

import PNP.Extraction
import Mathlib.Computability.NP

open SAT

/-- Théorème de Cook‑Levin (importé d'une bibliothèque formelle). -/
axiom cook_levin : ∀ (P : NPProblem), ∃ f : Instance P → SATInstance,
    ∀ x, P.decide x ↔ ∃ y, satisfies (f x) y

/-- SAT est dans P. -/
theorem SAT_in_P : ∃ (alg : ∀ n, SATInstance n → Option (Config n)),
    ∀ n inst, (∃ x, satisfies inst x) ↔ (alg n inst).isSome ∧ satisfies inst (alg n inst).get :=
  ⟨λ n inst => let χ := required_bond_dim n; let T := required_iterations n;
     if (power_iteration n inst 1 (by norm_num) T χ).some? then
       some (extract_max (power_iteration n inst 1 (by norm_num) T χ))
     else none,
   by
     intro n inst
     constructor
     · intro h_sol
       have h_correct := extraction_correct n inst 1 (by norm_num) (required_bond_dim n) (required_iterations n) (by linarith) (by linarith)
       simp [power_iteration] at h_correct
       exact ⟨h_sol, by simp⟩
     · intro ⟨hSome, hSat⟩
       exact hSat⟩

/-- P = NP. -/
theorem P_equals_NP : P_class = NP_class :=
  by
    have h_SAT_in_P : SAT ∈ P_class := ⟨SAT_in_P⟩
    have h_SAT_NP_complete : ∀ P ∈ NP_class, P ≤_p SAT :=
      by
        intro P hP
        obtain ⟨f, hf⟩ := cook_levin P
        exact ⟨f, hf⟩
    apply complexity_class_equality_from_complete_problem h_SAT_NP_complete h_SAT_in_P
