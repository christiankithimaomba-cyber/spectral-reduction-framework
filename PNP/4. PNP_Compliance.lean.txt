/-- Code de Gray récursif pour l'hypercube de dimension n. -/
def gray_code : ℕ → List (Fin n → Bool)
  | 0 => [fun _ => false]
  | m+1 =>
      let prev := gray_code m
      let flip_last (c : Fin m → Bool) (b : Bool) : Fin (m+1) → Bool :=
        fun i => if h : i.val = m then b else c ⟨i.val, by linarith [i.isLt]⟩
      (prev.map (flip_last · false)) ++ (prev.reverse.map (flip_last · true))

lemma gray_code_nodup (n : ℕ) : (gray_code n).Nodup := by
  induction n with
  | zero => simp [gray_code]
  | succ n ih =>
    simp [gray_code]
    refine List.Nodup.append (List.Nodup.map (fun _ _ => by simp) ih) ?_ ?_
    · apply List.Nodup.map
      · intro x y hxy
        simp [flip_last] at hxy
        ext i
        by_cases hi : i = Fin.last n
        · simp [hi, hxy]
        · simp [hi, hxy, Fin.castSucc]
      · exact ih.reverse
    · apply List.disjoint_of_forall_ne
      intro x hx y hy hxy
      simp at hx hy
      obtain ⟨a, ha, rfl⟩ := hx
      obtain ⟨b, hb, rfl⟩ := hy
      have : a = b := by ext i; by_cases hi : i = Fin.last n <;> simp_all
      contradiction

lemma gray_code_length (n : ℕ) : (gray_code n).length = 2^n := by
  induction n with
  | zero => simp [gray_code]
  | succ n ih => simp [gray_code, ih, pow_succ]

lemma gray_code_adjacent_junction (n : ℕ) :
    (hypercube_graph (n+1)).Adj
      ((gray_code (n+1)).get ⟨(2^n)-1, by linarith [pow_pos (by norm_num : 0 < 2) n]⟩)
      ((gray_code (n+1)).get ⟨2^n, by linarith [pow_pos (by norm_num : 0 < 2) n]⟩) :=
  by
    simp [gray_code]
    let len := 2^n
    have h_last : (gray_code n).last = (gray_code n).reverse.head := by simp
    simp [List.get_append_right, List.get_append_left]
    apply hypercube_adj_iff_one_bit.mpr
    use n
    constructor
    · simp [flip_last]
    · intro j hj
      simp [flip_last, hj]

lemma gray_code_adjacent (n : ℕ) (i : ℕ) (h : i < (gray_code n).length - 1) :
    (hypercube_graph n).Adj ((gray_code n).get ⟨i, by linarith⟩) ((gray_code n).get ⟨i+1, by linarith⟩) :=
  by
    induction n generalizing i with
    | zero => linarith
    | succ n ih =>
      simp [gray_code] at *
      let len := 2^n
      have hlen : len = (gray_code n).length := gray_code_length n
      by_cases h1 : i < len
      · have h_adj := ih i (by linarith)
        simp [List.get_append_left, List.get_map, h1] at h_adj ⊢
        exact h_adj
      · by_cases h2 : i = len - 1
        · simp [h2]
          exact gray_code_adjacent_junction n
        · have h3 : i > len := by linarith
          have h4 : i - len < len - 1 := by linarith
          have h_adj := ih (i - len) h4
          simp [List.get_append_right, List.get_map_rev, h3, hlen] at h_adj ⊢
          exact h_adj

/-- Structure de conformité pour le théorème de Brandão‑Horodecki. -/
structure BrandaoHorodeckiAssumptions (V : Type*) [Fintype V] [DecidableEq V] (P : SpectralProblem V) where
  hamiltonian_path : List V
  hamiltonian_path_nodup : hamiltonian_path.Nodup
  hamiltonian_path_length : hamiltonian_path.length = Fintype.card V
  hamiltonian_path_adj : ∀ i, i < hamiltonian_path.length - 1 → P.graph.Adj (hamiltonian_path[i]) (hamiltonian_path[i+1])
  interaction_range_le_one : ∀ i j, P.graph.Adj i j → (1 : ℕ) ≥ 1
  spectral_gap_min : ∃ γ : ℝ, 0 < γ ∧ spectral_gap (Hamiltonian P) ≥ γ
