RiemannExtraction.lean – Algorithme d'extraction.

import .RiemannPillars
import Kernel.PowerIteration

open SpectralLibrary

def extract_violation : Option (Assignment riemann_SAT) :=
  drsp_solver H_riemann

theorem extraction_correct :
    Satisfiable riemann_SAT → (extract_violation).isSome ∧
      satisfies riemann_SAT (extract_violation).get :=
  drsp_solver_correct H_riemann
