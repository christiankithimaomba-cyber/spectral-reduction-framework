/-
NConjectureHamiltonian.lean – Spectral Hamiltonian for the n-conjecture SAT instance,
and verification of the four pillars (P1–P4).
-/

import .NConjectureSAT
import Kernel.SpectralLibrary
import Kernel.KithimaBridge
import Kernel.MPSAreaLaw
import Kernel.HilbertCurve
import Kernel.Renormalisation

open SpectralLibrary

variable (n : ℕ) (ε : ℝ) (hε : ε > 0)

def Φ : SATInstance := n_conjecture_SAT n ε hε
def M : ℕ := Φ.numVars
def Config := Fin M → Bool

def V (σ : Config) : ℝ :=
  if satisfies Φ σ then 0 else M^2

def Δ : Matrix Config Config ℝ := adjacencyMatrixOf (hypercube_graph M)

def H_nε : Matrix Config Config ℝ := diagonal V + Δ

lemma H_irreducible : Irreducible H_nε :=
  matrix_is_irreducible_of_adjacency_graph (hypercube_connected M)

-- P1: unique strictly positive ground state (Perron‑Frobenius)
theorem ground_state_unique_pos :
    ∃! ψ : Config → ℝ,
      (∀ σ, ψ σ > 0) ∧ (∑ σ, ψ σ ^ 2 = 1) ∧ (H_nε *ᵥ ψ = (λ_min H_nε) • ψ) :=
  perron_frobenius (mk_spectral_problem H_nε)

-- P2: constant spectral gap (Kithima Bridge)
theorem spectral_gap_constant : spectral_gap H_nε ≥ 2 :=
  kithima_bridge (diagonal V) (by simp [V, M]; intro; split_ifs <;> linarith)
                Δ (by simp [Δ]; intros; split_ifs <;> linarith)
                1 (by norm_num) (HypercubeHarper.spectral_gap_adjacency (by linarith))

-- P3: logarithmic area law (via Brandão‑Horodecki + Hilbert curve + renormalisation)
theorem area_law_nε : ∃ C > 0, ∀ B : Finset Config,
    entanglement_entropy (reduced_density (ground_state H_nε) B) ≤ C * Real.log M :=
  renormalisation_area_law (by sorry) (hilbert_curve_bound (clause_graph Φ) (by exact bounded_degree))

-- P4: D‑RSP extraction (correctness from kernel)
theorem drsp_algorithm_nε : ∃ alg, polynomial_time alg ∧
    (∀ σ, satisfies Φ σ ↔ alg returns σ) :=
  drsp_correct H_nε
