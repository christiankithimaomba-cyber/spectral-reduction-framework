/- XVIII_GoormaghtighConfig.lean - Espace de configuration, potentiel, hamiltonien. -/

import Mathlib.Data.Fin.Basic
import Mathlib.Data.Nat.Basic
import Mathlib.Data.Fintype.Basic
import Mathlib.Combinatorics.SimpleGraph.Hypercube
import Kernel.SpectralLibrary

open Matrix

namespace SeriesXVIII.Goormaghtigh

/-- Paramètres de l'équation : deux entiers (m,n) avec 3 ≤ m ≤ n. -/
structure Params where
  m : ℕ
  n : ℕ
  hm : m ≥ 3
  hn : n ≥ 3
  hm_le_n : m ≤ n

variable (params : Params) (B : ℕ) (hB : B ≥ 1)

/-- Configuration : deux entiers x et y dans [1, B]. -/
def Config := Fin (B+1) × Fin (B+1)   -- indices 0..B, mais on utilisera 1..B

/-- Nombre total de bits : 2 * ceil(log2(B+1)). -/
def N := 2 * (Nat.log2 (B+1) + 1)

/-- Graphe sous-jacent : hypercube de dimension N. -/
def graph : SimpleGraph (Config B) := SimpleGraph.hypercube (Fin N)

lemma graph_connected : graph.Connected := SimpleGraph.hypercube_connected _

/-- Repunit : (x^e - 1)/(x - 1). -/
def repunit (x e : ℕ) : ℕ := (x^e - 1) / (x - 1)

/-- Prédicat de solution de Goormaghtigh. -/
def is_solution (σ : Config B) : Prop :=
  let x := σ.1.val; let y := σ.2.val;
  x > 1 ∧ y > 1 ∧ x ≠ y ∧
  repunit x params.m = repunit y params.n

/-- Décidabilité (quantification finie). -/
instance is_solution_decidable : Decidable (is_solution σ) :=
  decidable_of_iff' (by infer_instance) Iff.rfl

/-- Potentiel diagonal : 0 pour une solution, N² sinon. -/
def potential (σ : Config B) : ℝ :=
  if is_solution σ then 0 else (N : ℝ)^2

/-- Matrice d’adjacence de l’hypercube. -/
def adjacency : Matrix (Config B) (Config B) ℝ := adjacencyMatrixOf graph

/-- Hamiltonien spectral H = V + Δ. -/
def hamiltonian : Matrix (Config B) (Config B) ℝ := diagonal potential + adjacency

/-- Instance de SpectralProblem pour utiliser le kernel. -/
def spectral_problem : Kernel.SpectralLibrary.SpectralProblem (Config B) :=
  { potential := potential
    graph := graph
    epsilon := 1
    heps := by norm_num
    connected := graph_connected }

end SeriesXVIII.Goormaghtigh
