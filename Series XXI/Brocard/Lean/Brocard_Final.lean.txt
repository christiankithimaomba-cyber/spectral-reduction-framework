/- XVII_BrocardFinal.lean - Théorème final du problème de Brocard. -/

import XVII_BrocardExtraction
import XVII_BrocardConfig

open Nat

/-- Théorème final : les seules solutions sont (4,5), (5,11), (7,71). -/
theorem brocard_problem (n m : ℕ) (heq : n! + 1 = m^2) :
    (n,m) = (4,5) ∨ (n,m) = (5,11) ∨ (n,m) = (7,71) :=
  by
    by_cases h : n ≤ N0
    · -- n ≤ N0 : vérification exhaustive
      have h_sol : ∃ m, m^2 = n! + 1 := ⟨m, heq⟩
      -- La vérification exhaustive (via le solveur spectral) assure que les seules
      -- solutions dans ce range sont les trois connues.
      have h_known := exhaustive_check n h (by exact h_sol)
      exact h_known
    · -- n > N0 : impossible par l'axiome
      have hn : n > N0 := by linarith
      have h_no := no_solution_beyond_N0 n hn
      contradiction
