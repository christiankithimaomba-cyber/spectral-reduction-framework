import .GoldbachAreaLaw
import .GoldbachAmplitudeGap
import Kernel.PowerIteration

open Kernel.PowerIteration

def goldbach_extract (N : ℕ) (λ ε : ℝ) (hε : 0 < ε) (T χ : ℕ) : Option ℕ :=
  let H := GoldbachHamiltonian N λ ε hε
  let ψ0 := MPS.uniform_state (2*N+1)
  let ψT := power_iteration H ψ0 χ T
  let vec := MPS.state_vector ψT
  let i_max := Finset.argmax vec Finset.univ
  if let some n := i_max then
    if Nat.Prime n && Nat.Prime (2*N - n) then some n else none
  else none

theorem goldbach_extraction_correct (N : ℕ) (λ ε : ℝ) (hε : 0 < ε) (δ : ℝ) (hδ : 0 < δ)
    (h_amp : ∃ p, Nat.Prime p ∧ Nat.Prime (2*N - p) ∧
        ∀ q, ¬(Nat.Prime q ∧ Nat.Prime (2*N - q)) → ψ0 H q ≤ ψ0 H p - δ) :
    ∃ T χ, ∀ t ≥ T, ∀ χ' ≥ χ,
      let p := goldbach_extract N λ ε hε t χ'
      p.isSome ∧ Nat.Prime p.get ∧ Nat.Prime (2*N - p.get) :=
  by
    let H := GoldbachHamiltonian N λ ε hε
    let ψ_true := ground_state (GoldbachSpectralProblem N λ ε hε)
    let ψ0_vec := MPS.state_vector (MPS.uniform_state (2*N+1))
    have h_overlap : ∑ x, ψ0_vec x * ψ_true x > 0 :=
      sum_pos (fun x _ => mul_pos (by positivity) (ψ0_pos H hε x)) (by simp)
    have h_gap := goldbach_spectral_gap N λ ε hε
    obtain ⟨T0, χ0, h_conv⟩ := power_iteration_converges H (by apply IsSymm.adjoint_symm)
        (λ_min H) (by linarith) h_gap ψ0_vec ψ_true h_overlap h_amp (δ/4) (by linarith)
    exact ⟨T0, χ0, h_conv⟩
