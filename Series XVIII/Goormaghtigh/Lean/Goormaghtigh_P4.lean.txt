import XVIII_GoormaghtighP3
import Kernel.PowerIteration
import Kernel.DRSP

open Kernel.SpectralLibrary

namespace SeriesXVIII.Goormaghtigh

/-- Axiome : écart d’amplitude pour l’hamiltonien de Goormaghtigh.
    Preuve par perturbation de Kato (article XVIII.5). -/
axiom goormaghtigh_amplitude_gap (params : Params) (B : ℕ) (hB : B ≥ 1) :
    ∃ μ > 0, let ψ₀ := ground_state (spectral_problem params B);
    (∃ σ₀ : Config B, is_solution params σ₀ ∧
      ∀ σ, ¬ is_solution params σ → ψ₀ σ₀ ≥ ψ₀ σ + μ) ∨
    (∀ σ, ¬ is_solution params σ → ψ₀ a_constant_amplitude)

def H := hamiltonian params B
def M := (Φ params B).numVars
def α : ℝ := M^2 + 2*M + 1
def A := α • 1 - H
def uniform_mps : MPS M 1 := MPS.uniform

def power_iteration_goormaghtigh (T χ : ℕ) : MPS M χ :=
  power_iteration A uniform_mps χ T

def extract_solution (ψ : MPS M χ) : Option (ℕ × ℕ) :=
  let bits := drsp ψ
  if bits.length < 2*L then none else
  let x := bits_to_nat (bits.take L)
  let y := bits_to_nat (bits.drop L)
  if x > 1 ∧ y > 1 ∧ x ≠ y ∧ repunit x params.m = repunit y params.n then some (x,y) else none

/-- Théorème de correction : pour toute paire d’exposants admissible, le solveur trouve une solution. -/
theorem goormaghtigh_extraction_correct (params : Params) (B : ℕ) (hB : B ≥ B_of params.m params.n) :
    ∃ T0 χ0, ∀ T ≥ T0, ∀ χ ≥ χ0,
      let ψ := power_iteration_goormaghtigh T χ
      let opt := extract_solution ψ
      opt.isSome :=
  by
    have h_amp := goormaghtigh_amplitude_gap params B (by linarith)
    have h_gap := goormaghtigh_spectral_gap params B
    have h_conv := power_iteration_converges A h_gap uniform_mps (ground_state (spectral_problem params B))
      (by positivity)
    let δ := (h_amp.choose) / 4
    obtain ⟨T0, χ0, h_conv'⟩ := h_conv δ (by linarith)
    exact ⟨T0, χ0, fun T hT χ hχ => drsp_generic_correctness h_amp (h_conv' T hT χ hχ)⟩

end SeriesXVIII.Goormaghtigh
