/-
YMHamiltonian.lean – Hamiltonien spectral pour Yang–Mills sur réseau.
-/

import .YMLattice
import Kernel.SpectralLibrary

open SpectralLibrary

variable (N : ℕ) (G : Finset (Matrix (Fin 3) (Fin 3) ℂ)) (M : ℕ)
variable (encode : (Link N → G) → (Fin M → Bool)) (h_inj : Function.Injective encode)

/-- Action de Wilson pour une configuration U. -/
def wilsonAction (U : Link N → G) : ℝ :=
  let β : ℝ := 2.0   -- constante de couplage
  -- Liste des plaquettes (simplifiée; en pratique, on utilise tous les carrés unitaires)
  let plaquettes := List.product (List.finRange N) (List.range 4)   -- à compléter
  let sum := ∑ p in plaquettes, (1 - (1/3) * (U p).trace.re)
  β * sum

/-- Potentiel profond: zéro sur les configurations de vide (wilsonAction = 0), M^2 sinon. -/
def deepPotential (U : Link N → G) : ℝ :=
  if wilsonAction U = 0 then 0 else (M : ℝ)^2

/-- Terme de brisure de symétrie pour garantir l'unicité de l'état fondamental. -/
def symmetryBreaking (U : Link N → G) : ℝ :=
  (rank (encode U) : ℝ) / (M^2)

/-- Potentiel total. -/
def totalPotential (U : Link N → G) : ℝ := deepPotential U + symmetryBreaking U

/-- Espace de configuration après encodage: hypercube. -/
def Config := Fin M → Bool

-- On étend le potentiel à tout l'hypercube: les configurations non atteintes ont potentiel M^2.
def V_potential (x : Config) : ℝ :=
  if h : ∃ U, encode U = x then totalPotential (Classical.choose h) else M^2

-- La matrice diagonale du potentiel.
def V_matrix : Matrix Config Config ℝ := diagonal V_potential

-- Laplacien de l'hypercube.
def Δ_matrix : Matrix Config Config ℝ := adjacencyMatrixOf (SimpleGraph.hypercube (Fin M))

-- Hamiltonien spectral.
def H_YM : Matrix Config Config ℝ := V_matrix + Δ_matrix

-- Irréductibilité: l'hypercube est connexe.
lemma H_YM_irreducible : Irreducible H_YM := by
  apply Matrix.isIrreducible_of_adjacency_graph
  convert hypercube_connected M
  ext i j
  simp [H_YM, Δ_matrix, adjacencyMatrixOf]
  rfl

-- Par Perron‑Frobenius (kernel), on obtient un état fondamental unique strictement positif.
noncomputable def ground_state_YM : Config → ℝ :=
  ground_state (mkSpectralProblem H_YM)   -- à définir dans le kernel

-- Le potentiel total est non‑négatif.
lemma totalPotential_nonneg (U : Link N → G) : 0 ≤ totalPotential U := by
  by_cases h : wilsonAction U = 0
  · simp [totalPotential, deepPotential, h, symmetryBreaking]; positivity
  · simp [totalPotential, deepPotential, h, symmetryBreaking]; have : M^2 > 0 := by positivity; linarith

-- La matrice V est non‑négative.
lemma V_nonneg : ∀ i, V_matrix i i ≥ 0 := by
  intro i
  simp [V_matrix, V_potential]
  split_ifs <;> linarith [totalPotential_nonneg]
