/- XXXIV_BSD_Extraction.lean - Extraction algorithmique du rang par D‑RSP. -/

import XXXIV_BSD_Reduction
import Kernel.SpectralLibrary
import Kernel.DRSP

open SpectralLibrary

variable (E : EllipticCurve ℚ) (N : ℕ) (hN : N ≥ effective_N0 E)

/-- Ensemble des caractères de conducteur ≤ N. -/
def chars : Finset (HeckeCharacter ℚ) := characters_up_to E N

/-- Prédicat "bon caractère" : λ_χ(1)=1 et (dR) et (PT). -/
def good_char (χ : HeckeCharacter ℚ) : Bool :=
  eigenvalue E χ 1 = 1 ∧ dR_compatible E χ ∧ PT_compatible E χ

/-- Circuit pour le seuil r. -/
def threshold_circuit (r : ℕ) : Circuit (Fin (Finset.card chars) → Bool) :=
  let bits : Fin (Finset.card chars) → Var := fun i => input i 1
  let goods : List (Circuit Bool) :=
    List.ofFn (fun i => if good_char (chars i) then bits i else constant false)
  let sum := adder_tree goods
  ge_circuit sum (constant r)

/-- Instance SAT pour le seuil r. -/
def Φ_r (r : ℕ) : SATInstance := tseitin (threshold_circuit E N r)

/-- Hamiltonien spectral pour le seuil r. -/
def H_r (r : ℕ) : Matrix (Fin (2^((Φ_r r).numVars))) (Fin (2^((Φ_r r).numVars))) ℝ :=
  spectral_hamiltonian (Φ_r r)

/-- Détermination du rang par recherche binaire. -/
def compute_rank : ℕ :=
  let M := Finset.card chars
  let rec binsearch (lo hi : ℕ) : ℕ :=
    if lo ≥ hi then lo
    else
      let mid := (lo + hi + 1) / 2
      if drsp_solver (H_r E N mid) ≠ none then binsearch mid hi
      else binsearch lo (mid-1)
  binsearch 0 M

/-- Lemme : satisfiabilité équivaut à existence d'au moins r bons caractères. -/
lemma sat_threshold_iff_rank_ge (r : ℕ) :
    Satisfiable (Φ_r E N r) ↔ (Finset.card (chars.filter good_char) ≥ r) :=
  by
    -- Preuve standard par circuit et Tseitin.
    exact threshold_circuit_correct (chars.filter good_char) r

/-- Corollaire : le solveur D‑RSP répond non‑none ssi au moins r bons caractères existent. -/
lemma solver_nonempty_iff_rank_ge (r : ℕ) :
    drsp_solver (H_r E N r) ≠ none ↔ (Finset.card (chars.filter good_char) ≥ r) :=
  by
    rw [← sat_iff_solver_nonempty, sat_threshold_iff_rank_ge]

/-- Correction de la recherche binaire. -/
theorem binary_search_correct :
    compute_rank E N hN = Finset.card (chars.filter good_char) :=
  by
    apply binary_search_property (solver_nonempty_iff_rank_ge E N hN) 0 (Finset.card chars)

/-- Théorème final : compute_rank retourne le rang de la courbe. -/
theorem compute_rank_correct :
    compute_rank E N hN = rank_E E :=
  by
    rw [binary_search_correct, rank_equals_count_compatible E N hN]
    congr; ext χ; simp [good_char, dR_compatible, PT_compatible, eigenvalue]
