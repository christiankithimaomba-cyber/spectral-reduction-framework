/-
KummerVandiverTransfer.lean – Opérateur de transfert spectral via la K‑théorie.
Références : Adams (1966), Quillen (1971), Voevodsky (2011).
-/

import .KummerVandiverDefs
import Mathlib.AlgebraicTopology.Spectrum.KTheory
import Mathlib.LinearAlgebra.Matrix.Basic

open KTheory

/-- Le groupe K₄(ℤ) est connu : ≅ ℤ/2 ⊕ ℤ/2 (Borel, 1974). -/
def K4Z : Type := Fin 2 × Fin 2

/-- Axiome : Adams operation ψ^k sur K₄(ℤ) pour k premier avec l'ordre du groupe.
    Référence : Adams, "On the groups J(X) – IV" (1966). -/
axiom adams_operator (k : ℕ) : K4Z →ₗ[ℤ] K4Z

/-- L'opérateur d'Adams est auto‑adjoint pour la base standard.
    Référence : Adams conjecture (Quillen, 1971). -/
axiom adams_selfadjoint (k : ℕ) : ∀ x y : K4Z, ⟪adams_operator k x, y⟫ = ⟪x, adams_operator k y⟫

/-- Les valeurs propres de ψ^k sont des entiers (conséquence de l'intégralité des matrices de représentation). -/
axiom adams_eigenvalues_integral (k : ℕ) (λ : ℝ) :
    λ ∈ spectrum (matrix_of_linear_map (adams_operator k)) → λ ∈ ℤ

/-- Plus petit générateur du groupe (ℤ/pℤ)^× (existence garantie par la primalité). -/
noncomputable def primitive_root_mod (p : ℕ) (hp : p.Prime ∧ p > 2) : ℕ :=
  (primitive_root p).get (by exact exists_primitive_root (p - 1))

/-- Transfer operator T_p = ψ^k - I pour un générateur k de (ℤ/pℤ)^×. -/
noncomputable def transfer_operator (p : ℕ) (hp : p.Prime ∧ p > 2) : Matrix (Fin 4) (Fin 4) ℝ :=
  let k := primitive_root_mod p hp
  let M := matrix_of_linear_map (adams_operator k)
  M - 1

/-- La matrice de transfert est auto‑adjunte. -/
lemma transfer_selfadjoint (p : ℕ) (hp : p.Prime ∧ p > 2) :
    (transfer_operator p hp)ᵀ = transfer_operator p hp :=
  by simp [transfer_operator, adams_selfadjoint, transpose_sub, transpose_one]

/-- Les valeurs propres de T_p sont entières. -/
lemma transfer_eigenvalues_integral (p : ℕ) (hp : p.Prime ∧ p > 2) (λ : ℝ) :
    λ ∈ spectrum (transfer_operator p hp) → λ ∈ ℤ :=
  adams_eigenvalues_integral _ λ
