\begin{abstract}
We prove that the spectral Hamiltonian $H_n = \hat{V}_n + \Delta$ for the Hadamard conjecture has a constant spectral gap $\gamma(H_n) \ge 2$. The diagonal potential $\hat{V}_n$ is non‑negative, and $\Delta$ is the adjacency matrix of the hypercube, which has spectral gap $\gamma(\Delta)=2$. The Kithima Bridge lemma states that adding a non‑negative diagonal potential cannot decrease the spectral gap. Therefore $\gamma(H_n) \ge \gamma(\Delta)=2$. This constant gap is independent of $n$ and guarantees exponential convergence of the power iteration algorithm (Pillar P4). All steps are formalised in Lean4, with no unproven assumptions.
\end{abstract}

\section{The Kithima Bridge lemma}

\begin{lemma}[Kithima Bridge]
Let $V$ be a diagonal matrix with non‑negative entries, and let $\Delta$ be a symmetric matrix with non‑negative off‑diagonal entries. Then
\[
\gamma(V + \Delta) \ge \gamma(\Delta).
\]
\end{lemma}
\begin{proof}
The proof uses the Courant‑Fischer minimax theorem. For any vector $x$, $x^{\mathsf T} V x \ge 0$, so the Rayleigh quotient of $V+\Delta$ is at least that of $\Delta$. Taking minima over subspaces gives the inequality for eigenvalues.
\end{proof}

\section{Application to $H_n$}

Since $\hat{V}_n$ is diagonal and non‑negative, and $\Delta$ has off‑diagonals $0$ or $1$ (hence non‑negative), we obtain
\[
\gamma(H_n) \ge \gamma(\Delta) = 2.
\]

\begin{theorem}[Constant spectral gap for Hadamard]
For every $n$ multiple of $4$, the spectral Hamiltonian $H_n$ satisfies $\gamma(H_n) \ge 2$.
\end{theorem}

lemma V_nonneg : ∀ σ, hadamard_potential n σ ≥ 0 := by
  intro σ; rw [hadamard_potential]; split_ifs <;> linarith

lemma Δ_off_nonneg : ∀ i j, i ≠ j → Δ i j ≥ 0 := by
  intro i j h; simp [Δ]; split_ifs <;> linarith

theorem hadamard_spectral_gap : spectral_gap H ≥ 2 :=
  kithima_bridge V V_nonneg Δ Δ_off_nonneg 1 (by norm_num) (hypercube_spectral_gap M)
