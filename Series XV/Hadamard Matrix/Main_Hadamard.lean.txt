/--
  Théorème final : Pour tout entier n multiple de 4, il existe une matrice de Hadamard d'ordre n.
  La preuve combine les quatre piliers : existence d'un état fondamental unique (P1),
  trou spectral constant (P2), loi d'aire logarithmique (P3) et extraction déterministe (P4).
-/
theorem hadamard_conjecture_proved (n : ℕ) (hn : n % 4 = 0) :
    ∃ H : Matrix (Fin n) (Fin n) ℤ,
      (∀ i j, H i j = 1 ∨ H i j = -1) ∧
      H * Hᵀ = n • 1 :=
  hadamard_conjecture n hn   -- défini dans XV_HadamardFinal.lean
