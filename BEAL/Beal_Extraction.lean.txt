/-
BealExtraction.lean – Algorithme d'extraction pour Beal.
-/

import .BealPillars
import Kernel.PowerIteration

open SpectralLibrary

/-- Algorithme : extrait une assignation satisfaisante. -/
def extract_beal_counterexample : Option (Assignment beal_SAT) :=
  drsp_solver H_beal

/-- Si l'instance est satisfiable, l'algorithme trouve une solution. -/
theorem extraction_correct :
    Satisfiable beal_SAT → (extract_beal_counterexample).isSome ∧
      satisfies beal_SAT (extract_beal_counterexample).get :=
  drsp_solver_correct H_beal
