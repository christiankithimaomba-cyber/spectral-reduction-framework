/-
YMConstantGap.lean – Preuve du trou spectral constant via le pont de Kithima.
-/

import .YMHamiltonian
import Kernel.DiscreteCheeger

open SpectralLibrary

variable (N : ℕ) (G : Finset (Matrix (Fin 3) (Fin 3) ℂ)) (M : ℕ)
variable (encode : (Link N → G) → (Fin M → Bool)) (h_inj : Function.Injective encode)

-- Le pont de Kithima donne une borne inférieure sur le trou spectral.
theorem spectral_gap_YM : spectral_gap H_YM ≥ 2 := by
  let Δ := Δ_matrix
  have h_gap_Δ : spectral_gap Δ = 2 := HypercubeHarper.spectral_gap_adjacency (by linarith)
  have h_gap := kithima_bridge V_matrix V_nonneg Δ (by
    simp [Δ, adjacencyMatrixOf]; intro i j h; simp [h]) 1 (by norm_num)
  rw [h_gap_Δ] at h_gap
  exact h_gap

-- Le trou spectral est uniforme (indépendant de la taille du réseau).
theorem spectral_gap_uniform : ∀ N G M encode h_inj, spectral_gap H_YM ≥ 2 :=
  fun _ _ _ _ _ => spectral_gap_YM
