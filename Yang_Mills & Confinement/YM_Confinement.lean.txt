import .YMHamiltonian
import Kernel.WilsonLoop   -- ajout

open SpectralLibrary

-- On suppose que la loi d'aire pour le lacet de Wilson est vraie (axiome).
-- Reference: area law for Wilson loop from constant spectral gap.
axiom area_law_Wilson_loop_holds : ∃ σ > 0, ∃ C > 0, ∀ R T : ℕ,
    |⟨W_{R×T}⟩_ψ₀| ≤ C * Real.exp (-σ * R * T)

-- Alors la conductance décroît exponentiellement.
lemma conductance_exp_decay (R : ℕ) : ∃ C' > 0, Φ(R) ≤ C' * Real.exp (-σ * R) :=
  area_law_to_conductance σ hσ C hC area_law_Wilson_loop_holds

-- Potentiel linéaire.
theorem linear_potential : ∃ σ > 0, ∀ R : ℝ, V_eff(R) = σ * R + O(1) :=
  exponential_decay_to_linear_potential σ hσ C' hC' (conductance_exp_decay)
