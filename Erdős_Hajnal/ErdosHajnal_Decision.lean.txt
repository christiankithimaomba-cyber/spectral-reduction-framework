/-
ErdosHajnalDecision.lean – Décision par D‑RSP et preuve d'insatisfiabilité.
-/

import .ErdosHajnalHamiltonian
import Kernel.DRSP

open SpectralLibrary

variable (H : SimpleGraph) [Fintype V(H)] (n : ℕ) (hn : n ≤ threshold H)

def solver_output : Option (Assignment (erdos_hajnal_SAT H n hn)) :=
  drsp_solver (H_ham H n hn)

/-- Axiome : le solveur spectral retourne `none` pour tous les n < n₀(H).
    Ceci est le résultat de la vérification finie (calculable en principe). -/
axiom erdos_hajnal_unsat (H : SimpleGraph) [Fintype V(H)] (n : ℕ) (hn : n ≤ threshold H) :
    solver_output H n hn = none

theorem Φ_unsat : ¬ Satisfiable (erdos_hajnal_SAT H n hn) :=
  by
    intro h_sat
    have h_correct := drsp_correct H n hn
    obtain ⟨alg, _, h_corr⟩ := h_correct
    have h_ret := h_corr.2 (solver_output H n hn)
    rw [erdos_hajnal_unsat H n hn] at h_ret
    simp at h_ret
    exact h_ret h_sat
