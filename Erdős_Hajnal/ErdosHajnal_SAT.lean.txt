/-
ErdosHajnalSAT.lean – Conversion du circuit en instance 3‑SAT.
-/

import .ErdosHajnalCircuit
import Kernel.Tseitin

open Circuit

variable (H : SimpleGraph) [Fintype V(H)] (n : ℕ) (hn : n ≤ threshold H)

def Φ : SATInstance := tseitin (counterexample_circuit H n hn)

theorem sat_correct :
    Satisfiable Φ ↔ ∃ G : SimpleGraph (Fin n), is_counterexample H n G :=
  by
    rw [tseitin_correct]
    exact counterexample_circuit_correct H n hn
