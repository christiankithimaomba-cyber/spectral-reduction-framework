/-
HutterSAT.lean – Convert the UTM circuit to a SAT instance.
-/

import .HutterUTM
import Kernel.Tseitin

open Circuit

variable (L T : ℕ) (x : TargetString)

/-- SAT instance: satisfiable iff there exists a program of length L that outputs x within T steps. -/
def hutter_SAT : SATInstance :=
  tseitin (utm_circuit L T x)

/-- Correctness: satisfiability ⇔ existence of a program. -/
theorem hutter_SAT_correct :
    Satisfiable (hutter_SAT L T x) ↔ ∃ p : Program, program_length p = L ∧ UTM_eval p T = some x :=
  by
    rw [tseitin_correct]
    exact utm_circuit_correct L T x
