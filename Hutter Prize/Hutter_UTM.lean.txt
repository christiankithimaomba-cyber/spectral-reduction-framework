/-
HutterUTM.lean – Boolean circuit simulating a universal Turing machine for a bounded number of steps.
Reference: standard construction in complexity theory.
-/

import .HutterDefs
import Kernel.Circuit.Basic

open Circuit

variable (L T : ℕ)   -- L: program length (bits), T: time bound

/-- Number of bits needed to encode the state of the UTM (states + tape). -/
def state_bits : ℕ := 10^6   -- fixed constant for a concrete UTM

/-- The UTM simulation circuit: inputs are the program bits (L bits),
    outputs 1 iff the UTM halts within T steps and produces the target string x. -/
axiom utm_circuit (x : TargetString) : Circuit (Fin L → Bool)

/-- Correctness of the UTM circuit. -/
axiom utm_circuit_correct (x : TargetString) (bits : Fin L → Bool) :
    (utm_circuit L T x).eval bits = true ↔
    let p := program_from_bits bits
    UTM_eval p T = some x
