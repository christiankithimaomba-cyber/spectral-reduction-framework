/-
HutterDefs.lean – Basic definitions for the Hutter Prize (optimal compression).
-/

import Mathlib.Data.String.Basic
import Mathlib.Data.Nat.Basic

/-- The target string to compress (e.g., enwik9). -/
def TargetString : Type := String

/-- A program is a binary string. -/
def Program := List Bool

/-- Length of a program. -/
def program_length (p : Program) : ℕ := p.length

/-- Universal Turing machine (UTM) simulation: given a program and a time bound,
    outputs the string produced after that many steps, or none if not finished. -/
axiom UTM_eval (p : Program) (T : ℕ) : Option TargetString

/-- The Kolmogorov complexity of a string x is the length of the shortest program
    that outputs x on the UTM. -/
def kolmogorov_complexity (x : TargetString) : ℕ :=
  Inf { p : Program | UTM_eval p (10^12) = some x }

/-- The Hutter Prize: find a program p that minimizes length(p) such that UTM_eval p T = x
    for a sufficiently large time bound T (e.g., T = 10^12). -/
def hutter_prize_solved (x : TargetString) : Prop :=
  ∃ p : Program, UTM_eval p (10^12) = some x ∧ ∀ q, UTM_eval q (10^12) = some x → program_length p ≤ program_length q
