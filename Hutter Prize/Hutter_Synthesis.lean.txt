/-
HutterSynthesis.lean – Final theorem: the Hutter Prize is solvable.
-/

import .HutterDecision

open SpectralLibrary

def char_to_bits (c : Char) : List Bool :=
  let n := c.toNat
  List.reverse (List.range 8 |>.map (fun i => (n >> i) % 2 == 1))

def trivial_program (x : TargetString) : Program :=
  x.toList.flatMap char_to_bits

axiom utm_copy_correct (x : TargetString) (T : ℕ) (hT : T ≥ 2 * x.length * 8 + 10) :
    UTM_eval (trivial_program x) T = some x

theorem hutter_prize_solved (x : TargetString) (T : ℕ) (hT : T ≥ 10^12) :
    ∃ p : Program, UTM_eval p T = some x ∧ ∀ q, UTM_eval q T = some x → program_length p ≤ program_length q :=
  by
    have h_exists : ∃ p, UTM_eval p T = some x :=
      ⟨trivial_program x, utm_copy_correct x T (by linarith)⟩
    let p := shortest_program T x
    exact shortest_program_correct T x h_exists
