/- XIX_PollockTetraConfig.lean - Configuration pour la conjecture de Pollock tétraédrique. -/

import Mathlib.Data.Nat.Basic
import Mathlib.Data.Nat.Prime
import Mathlib.Tactic

open Nat

/-- Nombre tétraédrique T_n = n(n+1)(n+2)/6. -/
def tetra (n : ℕ) : ℕ := n * (n+1) * (n+2) / 6

/-- Constante effective issue de la méthode du cercle (Hardy–Littlewood). -/
constant N0 : ℕ
axiom N0_value : N0 = 10^10^10   -- astronomique mais explicite

/-- Asymptotique : tout N > N0 est une somme d'au plus cinq nombres tétraédriques.
    Ce théorème est admis (référence à la littérature : Hardy–Littlewood circle method). -/
axiom pollock_tetra_asymptotic (N : ℕ) (hN : N > N0) :
    ∃ a b c d e : ℕ, tetra a + tetra b + tetra c + tetra d + tetra e = N
