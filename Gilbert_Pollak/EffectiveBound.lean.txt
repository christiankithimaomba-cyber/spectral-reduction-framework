/-
EffectiveBound.lean – Effective asymptotic bound for the Gilbert–Pollak conjecture.
Reference: Du–Hwang (1992), Chung–Gilbert (1976).
-/

import Mathlib.Data.Nat.Basic

/-- Explicit constant N0 such that for all point sets with more than N0 points,
    the inequality holds. -/
noncomputable def N0 : ℕ := 10^10^10

/-- Axiom: asymptotic bound (Du–Hwang, 1992). -/
axiom gp_asymptotic (P : Finset (ℝ × ℝ)) (h : Fintype.card P > N0) :
    SMT_length P ≥ ρ * MST_length P
