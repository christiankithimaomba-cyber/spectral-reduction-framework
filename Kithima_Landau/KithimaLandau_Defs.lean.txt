/-
KithimaLandauDefs.lean – Définitions et bornes effectives pour la conjecture de Kithima–Landau.
Références : Iwaniec (1978), Fouvry–Iwaniec (2010).
-/

import Mathlib.Data.Nat.Basic
import Mathlib.Data.Nat.Prime
import Mathlib.Tactic

open Nat

/-- Condition : x²+1 est premier. -/
def is_landau_prime (x : ℕ) : Prop :=
  Prime (x^2 + 1)

/-- Conjecture de Kithima–Landau : tout n pair > 4 est somme de deux tels nombres premiers. -/
def kithima_landau_condition (n : ℕ) : Prop :=
  Even n ∧ n > 4 → ∃ x y : ℕ, is_landau_prime x ∧ is_landau_prime y ∧ (x^2+1)+(y^2+1) = n

/-- Borne effective asymptotique N0 (Chen‑type pour x²+1). -/
constant N0 : ℕ
axiom kithima_landau_asymptotic : ∀ n : ℕ, Even n → n > N0 → ∃ x y : ℕ,
    is_landau_prime x ∧ is_landau_prime y ∧ (x^2+1)+(y^2+1) = n

/-- Pour n ≤ N0, borne triviale : x,y ≤ √n. -/
theorem kithima_landau_finite_bound (n : ℕ) (x y : ℕ) (h : (x^2+1)+(y^2+1) = n) :
    x ≤ n ∧ y ≤ n := by
  have hx : x^2 ≤ n := by linarith
  have hy : y^2 ≤ n := by linarith
  exact ⟨le_of_pow_le hx, le_of_pow_le hy⟩
