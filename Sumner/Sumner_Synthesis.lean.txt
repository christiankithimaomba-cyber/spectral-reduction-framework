/-
SumnerSynthesis.lean – Final proof of Sumner's conjecture.
-/

import .SumnerDecision
import .SumnerAsymptotic

open SimpleGraph

theorem sumner_conjecture (n : ℕ) (hn : n ≥ 3) :
    ∀ (T : Tournament (2*n - 2)) (F : OrientedTree n),
      ∃ (f : Fin n → Fin (2*n - 2)), Function.Injective f ∧
        ∀ (u v : Fin n), F.Adj u v → T.Adj (f u) (f v) :=
  by
    by_cases h : n < N0
    · -- small n: use spectral verification
      intro T F
      by_contra h_contr
      have h_no_counter := no_counterexample n h
      exact h_no_counter T F h_contr
    · -- large n: use asymptotic bound
      have h_large : n ≥ N0 := by linarith
      exact sumner_asymptotic n h_large
