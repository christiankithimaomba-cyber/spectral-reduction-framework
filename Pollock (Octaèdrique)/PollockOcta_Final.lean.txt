/- XX_PollockOctaFinal.lean - Théorème final de la conjecture de Pollock octaédrique. -/

import XX_PollockOctaExtraction
import XX_PollockOctaConfig

open Nat

/-- Pour tout N ≥ 1, il existe sept nombres octaédriques dont la somme est N. -/
theorem pollock_octahedral_conjecture (N : ℕ) (hN : N ≥ 1) :
    ∃ a b c d e f g : ℕ, octahedral a + octahedral b + octahedral c + octahedral d +
                         octahedral e + octahedral f + octahedral g = N :=
  by
    by_cases h : N ≤ N0
    · exact octa_decomposition_exists N h
    · have h_gt : N > N0 := by linarith
      exact pollock_octa_asymptotic N h_gt
