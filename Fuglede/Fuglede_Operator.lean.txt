/- XXXV_Fuglede_Operator.lean - Opérateur spectral U_Ω et groupe unitaire. -/

import XXXV_Fuglede_Config
import Mathlib.Analysis.Sobolev
import Mathlib.OperatorTheory.StrongOperatorTopology

variable {d : ℕ} (hd : 1 ≤ d ∧ d ≤ 4) (Ω : Set ℝ^d) (hΩ : MeasurableSet Ω ∧ Bounded Ω ∧ LipschitzBoundary Ω)

/-! ### Définition de l'opérateur U_Ω -/

/-- Opérateur U_Ω sur L^2(Ω) : i ∂/∂x₁ avec conditions aux limites périodiques. -/
axiom U_Ω : L^2 Ω →ₗ[ℝ] L^2 Ω

/-- Domaine dense d'opérateur. -/
axiom U_Ω_dense : DenseSubspace (U_Ω.domain)

/-- Auto‑adjoint. -/
axiom U_Ω_selfadjoint : IsSelfAdjoint U_Ω

/-- Résolvante compacte (spectre discret). -/
axiom U_Ω_compact_resolvent : CompactResolvent U_Ω

/-- Base orthonormée de vecteurs propres avec valeurs propres croissantes. -/
axiom U_Ω_eigenbasis : ∃ (φ : ℕ → L^2 Ω) (λ : ℕ → ℝ),
  OrthonormalBasis φ ∧ ∀ n, U_Ω (φ n) = λ n • φ n ∧ Tendsto λ atTop atTop

/-! ### Groupe unitaire exponentiel -/

/-- Groupe unitaire à un paramètre (dans la direction x₁). -/
def exp_U (t : ℝ) : L^2 Ω →ₗ[ℝ] L^2 Ω :=
  spectral_exponential (U_Ω t)   -- défini par le théorème spectral

/-- Groupe unitaire multi‑paramètres (t ∈ ℝ^d). -/
def exp_group (t : ℝ^d) : L^2 Ω →ₗ[ℝ] L^2 Ω :=
  let t1 := t 0   -- première coordonnée
  let t_rest := (fun i => t i) (Fin.tail)
  exp_U t1 ∘ exp_group_rest t_rest   -- récursif

/-- Le groupe est unitaire. -/
axiom exp_group_unitary (t : ℝ^d) : IsUnitary (exp_group t)

/-- Le groupe est fortement continu. -/
axiom exp_group_strong_continuous : ∀ f : L^2 Ω, Continuous (fun t => exp_group t f)

/-- Propriété de générateur : dérivée en t=0 donne i ∇. -/
axiom generator_property (f : L^2 Ω) (hf : f ∈ smooth_compactly_supported_interior Ω) :
  deriv (fun t => exp_group t f) 0 = i • gradient f
