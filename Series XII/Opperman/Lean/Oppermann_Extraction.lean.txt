/-
OppermannExtraction.lean – Algorithme d'extraction pour le range fini.
-/

import .OppermannPillars
import Kernel.DRSP

open SpectralLibrary

variable (n : ℕ) (hn : n ≥ 1)

def extract_pair : Option (ℕ × ℕ) :=
  let H := H_opp n hn
  match drsp_solver H with
  | none => none
  | some σ => some (decode_pair σ)

theorem extraction_correct (n : ℕ) (hn : n ≥ 1) (h_le : n ≤ N0) :
    extract_pair n hn ≠ none :=
  by
    -- Pour n ≤ N0, l'existence d'une solution est garantie par vérification exhaustive
    have h_sat : Satisfiable (oppermann_SAT n hn) :=
      oppermann_finite_verification n hn h_le
    exact drsp_solver_correct (H_opp n hn) h_sat

/-- Axiome : pour n ≤ N0, il existe une solution (vérification exhaustive). -/
axiom oppermann_finite_verification (n : ℕ) (hn : n ≥ 1) (h_le : n ≤ N0) :
    oppermann_condition n
