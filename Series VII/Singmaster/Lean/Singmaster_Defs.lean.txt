/-
SingmasterDefs.lean – Définitions et bornes effectives pour la conjecture de Singmaster.
Références : Shorey–Tijdeman (1986), de Weger (1989).
-/

import Mathlib.Data.Nat.Basic
import Mathlib.Data.Nat.Choose
import Mathlib.Data.Nat.Prime
import Mathlib.Tactic

open Nat

/-- La multiplicité m(t) : nombre de solutions de C(n,k)=t avec 1 ≤ k ≤ n/2. -/
def multiplicity (t : ℕ) : ℕ :=
  Finset.card (Finset.filter (fun (n,k) => choose n k = t ∧ 1 ≤ k ∧ 2*k ≤ n)
    (Finset.product (Finset.Icc 0 t) (Finset.Icc 0 t)))

/-- Seuil explicite T0 (de Weger, 1989). -/
constant T0 : ℕ
axiom deweger_threshold : ∀ t > T0, ∀ n k : ℕ, 1 ≤ k ≤ n/2 → choose n k = t → (k = 1 ∨ k = n-1)

/-- Pour t ≤ T0, tous les solutions satisfont n,k ≤ t (borne triviale). -/
theorem bound_small (t : ℕ) (ht : t ≤ T0) (n k : ℕ) (hk : 1 ≤ k) (hkn : k ≤ n/2)
    (heq : choose n k = t) : n ≤ t ∧ k ≤ t := by
  have h1 : n ≤ t := by
    by_cases hk1 : k = 1
    · rw [choose_one] at heq; linarith
    · have : choose n k ≥ n := choose_le_n n k (by linarith)
      linarith
  exact ⟨h1, le_trans hk (by linarith)⟩
