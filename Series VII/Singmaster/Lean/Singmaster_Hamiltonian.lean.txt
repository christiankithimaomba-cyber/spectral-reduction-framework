/-
SingmasterHamiltonian.lean – Hamiltonien spectral pour l'instance SAT.
-/

import .SingmasterSAT
import Kernel.SpectralLibrary

open SpectralLibrary

variable (t : ℕ) (ht : t ≤ T0)

def Φ : SATInstance := singmaster_SAT t ht
def M : ℕ := Φ.numVars
def Config := Fin M → Bool

def V (σ : Config) : ℝ :=
  if satisfies Φ σ then 0 else M^2

def Δ : Matrix Config Config ℝ := adjacencyMatrixOf (hypercube_graph M)

def H_singmaster : Matrix Config Config ℝ := diagonal V + Δ

lemma H_irreducible : Irreducible H_singmaster :=
  matrix_is_irreducible_of_adjacency_graph (hypercube_connected M)
