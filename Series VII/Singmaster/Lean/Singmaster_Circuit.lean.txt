/-
SingmasterCircuit.lean – Circuit booléen pour tester C(n,k)=t.
-/

import .SingmasterDefs
import Kernel.Circuit.Basic
import Kernel.Circuit.Arithmetic

open Circuit

variable (t : ℕ) (ht : t ≤ T0)

def L : ℕ := Nat.log2 t + 1

/-- Circuit pour le produit n(n-1)...(n-k+1) pour un k fixé. -/
def numerator_product (n : Circuit (List Bool)) (k : ℕ) : Circuit (List Bool) :=
  let factors := List.range k |>.map (fun i => sub n (constant i))
  factors.foldl (fun acc f => mul acc f) (constant 1)

/-- Circuit complet pour tester C(n,k)=t. -/
def binom_circuit : Circuit (Fin (2*L) → Bool) :=
  let n := input 0 L
  let k := input L L
  let cases : List (Circuit Bool) :=
    List.range 1 (t+1) |>.map (fun k0 =>
      let num := numerator_product n k0
      let den := constant (Nat.factorial k0)
      let quot := div num den
      let eq := eq quot (constant t)
      let k_eq := eq k (constant k0)
      let le := le (mul (constant 2) k) n
      and (and k_eq eq) le)
  cases.foldl (fun acc c => or acc c) (constant false)

/-- Correction du circuit (axiome, car preuve standard). -/
axiom binom_circuit_correct (bits : Fin (2*L) → Bool) :
    (binom_circuit t).eval bits = true ↔
    let n := bits_to_nat (bits.slice 0 L)
    let k := bits_to_nat (bits.slice L L)
    1 ≤ k ∧ 2*k ≤ n ∧ Nat.choose n k = t
