/- XIX_PollockTetraFinal.lean - Théorème final de la conjecture de Pollock tétraédrique. -/

import XIX_PollockTetraExtraction
import XIX_PollockTetraConfig

open Nat

/-- Pour tout N ≥ 1, il existe cinq nombres tétraédriques dont la somme est N. -/
theorem pollock_tetrahedral_conjecture (N : ℕ) (hN : N ≥ 1) :
    ∃ a b c d e : ℕ, tetra a + tetra b + tetra c + tetra d + tetra e = N :=
  by
    by_cases h : N ≤ N0
    · exact tetra_decomposition_exists N h
    · have h_gt : N > N0 := by linarith
      exact pollock_tetra_asymptotic N h_gt
