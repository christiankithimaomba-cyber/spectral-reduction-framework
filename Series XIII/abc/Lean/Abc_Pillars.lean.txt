import .AbcHamiltonian
import Kernel.KithimaBridge
import Kernel.BrandaoHorodecki
import Kernel.HilbertCurve
import Kernel.MPSAreaLaw

open SpectralLibrary

variable (N0 : ℕ) (hN0 : N0 > 0)

-- P1
lemma ground_state_unique_pos :
    ∃! ψ : Config → ℝ, (∀ σ, ψ σ > 0) ∧ (∑ σ, ψ σ ^ 2 = 1) ∧
      (H_abc N0 hN0 *ᵥ ψ = (λ_min (H_abc N0 hN0)) • ψ) :=
  perron_frobenius (mk_spectral_problem (H_abc N0 hN0))

-- P2 : trou spectral constant
lemma spectral_gap_constant : spectral_gap (H_abc N0 hN0) ≥ 2 :=
  kithima_bridge (diagonal V) (by simp [V]; intro; split_ifs <;> linarith)
                Δ (by simp [Δ]; intros; split_ifs <;> linarith)
                1 (by norm_num) (HypercubeHarper.spectral_gap_adjacency (by linarith))

-- P3 : loi d'aire logarithmique
lemma area_law_abc : ∃ C > 0, ∀ B : Finset Config,
    entanglement_entropy (reduced_density (ground_state (mk_spectral_problem H_abc)) B) ≤ C * Real.log M :=
  area_law_for_hypercube (spectral_gap_constant N0 hN0)

-- P4 : extraction D‑RSP
theorem drsp_correct_abc :
    ∃ alg, polynomial_time alg ∧
      (Satisfiable (abc_SAT N0) ↔ (alg N0).isSome ∧ satisfies _ (alg N0).get) :=
  drsp_correct (H_abc N0 hN0)
