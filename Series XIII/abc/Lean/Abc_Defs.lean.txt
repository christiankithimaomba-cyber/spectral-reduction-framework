/-
AbcDefs.lean – Définitions et bornes effectives pour la conjecture abc.
Références : Stewart–Yu (2001), Matveev (2000).
-/

import Mathlib.Data.Nat.Basic
import Mathlib.Data.Nat.Prime
import Mathlib.Data.Nat.GCD
import Mathlib.NumberTheory.Radical
import Mathlib.Tactic

open Nat

/-- Radical d'un entier : produit des facteurs premiers distincts. -/
def rad (n : ℕ) : ℕ := n.rad

/-- Condition de contre‑exemple pour ε=1 : a+b=c, gcd(a,b,c)=1, c > C_eps * rad(abc)^2. -/
def counterexample (a b c C_eps : ℕ) : Prop :=
  a + b = c ∧ gcd a (gcd b c) = 1 ∧ c > C_eps * (rad (a*b*c))^2

/-- Constante explicite C(1) (on peut prendre 1, mais la littérature donne une valeur). -/
constant C_eps : ℕ
axiom C_eps_pos : C_eps > 0

/-- Borne effective N0 (Stewart–Yu) : tout contre‑exemple avec a,b,c ≤ N0? Non, la borne dit que si un contre‑exemple existe, alors a,b,c ≤ N0. -/
constant N0 : ℕ
axiom stewart_yu_bound (a b c : ℕ) (h : counterexample a b c C_eps) : a ≤ N0 ∧ b ≤ N0 ∧ c ≤ N0

/-- Pour la vérification finie, on utilise la borne triviale que a,b,c ≤ c. -/
theorem trivial_bound (a b c : ℕ) (hsum : a + b = c) : a ≤ c ∧ b ≤ c := by
  constructor <;> linarith
