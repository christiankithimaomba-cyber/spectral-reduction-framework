import Kernel.SpectralLibrary
import Mathlib.GroupTheory.Perm.Basic
import Mathlib.Combinatorics.SimpleGraph.Cayley

open SpectralLibrary

def Perm (n : ℕ) := Equiv.Perm (Fin n)

def permutation_graph (n : ℕ) : SimpleGraph (Perm n) :=
  let gen := List.map (fun i => Equiv.swap i (i+1)) (List.range (n-1))
  CayleyGraph (Perm n) gen

-- Axiom: Le graphe des transpositions adjacentes est connexe (groupe symétrique engendré par les transpositions)
-- Reference: Mathlib.GroupTheory.Perm.Basic
axiom permutation_graph_connected (n : ℕ) : (permutation_graph n).Connected

structure Graph (n : ℕ) where
  adj : Fin n → Fin n → Bool
  symm : ∀ i j, adj i j = adj j i
  irref : ∀ i, ¬ adj i i

def is_hamiltonian_cycle (G : Graph n) (σ : Perm n) : Bool :=
  ∀ i : Fin n, G.adj (σ i) (σ (i+1)) where
    i+1 := if i.val = n-1 then 0 else i+1

def hamiltonian_potential (G : Graph n) (σ : Perm n) : ℝ :=
  if is_hamiltonian_cycle G σ then 1 else 0

def hamiltonian_problem (G : Graph n) (ε : ℝ) (hε : 0 < ε) : SpectralProblem (Perm n) where
  potential := hamiltonian_potential G
  graph := permutation_graph n
  epsilon := ε
  heps := hε
  connected := permutation_graph_connected n
