/-
EdgeMatchingEncoding.lean – Encoding of placements as binary strings and validity circuit.
-/

import .EdgeMatchingDefs
import Kernel.Circuit.Basic
import Kernel.Circuit.Arithmetic

open Circuit

variable (pieces : List Piece) (rows cols : ℕ)

def num_pieces : ℕ := pieces.length
def total_cells : ℕ := rows * cols
def bits_per_cell : ℕ := Nat.log2 total_cells + 1
def bits_per_orientation : ℕ := 2   -- 4 orientations
def total_bits : ℕ := num_pieces * (bits_per_cell + bits_per_orientation)

/-- Decode a configuration into a placement. -/
axiom decode_placement (σ : Fin total_bits → Bool) : Placement num_pieces rows cols

/-- Circuit that tests validity of a placement. -/
axiom validity_circuit : Circuit (Fin total_bits → Bool)

/-- Correctness of the validity circuit. -/
axiom validity_circuit_correct (σ : Fin total_bits → Bool) :
    (validity_circuit).eval σ = true ↔ valid_placement pieces rows cols (decode_placement σ)

/-- SAT instance for existence of a valid placement. -/
def edge_matching_SAT : SATInstance := tseitin validity_circuit

/-- Satisfiability iff there exists a valid placement. -/
theorem edge_matching_SAT_correct :
    Satisfiable edge_matching_SAT ↔ ∃ p : Placement num_pieces rows cols, valid_placement pieces rows cols p :=
  by
    rw [tseitin_correct]
    exact validity_circuit_correct
