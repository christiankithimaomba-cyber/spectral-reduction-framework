/-
EdgeMatchingSynthesis.lean – Final theorem: edge matching puzzles are solvable.
-/

import .EdgeMatchingExtraction

open SpectralLibrary

theorem edge_matching_puzzle_solved (pieces : List Piece) (rows cols : ℕ) (h : rows * cols = pieces.length) :
    ∃ p : Placement pieces.length rows cols, valid_placement pieces rows cols p :=
  ⟨extract_placement pieces rows cols, edge_matching_solved pieces rows cols h⟩
