/-
EdgeMatchingHamiltonian.lean – Spectral Hamiltonian and verification of the four pillars.
-/

import .EdgeMatchingEncoding
import Kernel.SpectralLibrary
import Kernel.KithimaBridge
import Kernel.ClusterExpansion
import Kernel.BrandaoHorodecki
import Kernel.HilbertCurve
import Kernel.Renormalisation

open SpectralLibrary

variable (pieces : List Piece) (rows cols : ℕ)

def Φ : SATInstance := edge_matching_SAT pieces rows cols
def M : ℕ := Φ.numVars
def Config := Fin M → Bool

def V (σ : Config) : ℝ := if satisfies Φ σ then 0 else M^2
def Δ : Matrix Config Config ℝ := adjacencyMatrixOf (hypercube_graph M)
def H_edge : Matrix Config Config ℝ := diagonal V + Δ

lemma H_irreducible : Irreducible H_edge :=
  matrix_is_irreducible_of_adjacency_graph (hypercube_connected M)

-- P1: unique strictly positive ground state
theorem ground_state_unique_pos :
    ∃! ψ : Config → ℝ, (∀ σ, ψ σ > 0) ∧ (∑ σ, ψ σ ^ 2 = 1) ∧
      (H_edge *ᵥ ψ = (λ_min H_edge) • ψ) :=
  perron_frobenius (mk_spectral_problem H_edge)

-- P2: constant spectral gap (Kithima Bridge)
theorem spectral_gap_constant : spectral_gap H_edge ≥ 2 :=
  kithima_bridge (diagonal V) (by simp [V, M]) Δ (by simp) 1 (by norm_num)

-- Auxiliary: clause graph has bounded degree
lemma clause_graph_bounded_degree : ∃ d, ∀ v, degree (clause_graph Φ) v ≤ d :=
  degree_reduction_bounded Φ

-- Exponential decay (from constant gap)
lemma exp_decay_edge (ψ := ground_state H_edge) :
    ∃ C ξ > 0, ∀ A B : Finset (Fin M), Disjoint A B →
      |⟨σ_A σ_B⟩_ψ - ⟨σ_A⟩_ψ * ⟨σ_B⟩_ψ| ≤ C * Real.exp (- (graph_distance (clause_graph Φ) A B) / ξ) :=
  cluster_expansion_correlations (clause_graph Φ) clause_graph_bounded_degree ψ spectral_gap_constant

-- Linear area law via Brandão‑Horodecki
lemma linear_area_law_edge (ψ := ground_state H_edge) :
    ∃ C1 > 0, ∀ B : Finset (Fin M), Connected B →
      entanglement_entropy (reduced_density ψ B) ≤ C1 * edge_boundary (clause_graph Φ) B :=
  brandao_horodecki (clause_graph Φ) clause_graph_bounded_degree ψ exp_decay_edge

-- Hilbert curve bound
lemma hilbert_bound_edge :
    ∃ (π : Fin M → Fin M) (C_hilb > 0), Bijective π ∧
      ∀ k, edge_boundary (clause_graph Φ) {π i | i < k} ≤ C_hilb * Real.log M :=
  hilbert_curve_bound (clause_graph Φ) clause_graph_bounded_degree

-- P3: logarithmic area law
theorem area_law_edge (ψ := ground_state H_edge) :
    ∃ C > 0, ∀ B : Finset Config,
      entanglement_entropy (reduced_density ψ (B.map (fun x => x))) ≤ C * Real.log M :=
  renormalisation_area_law linear_area_law_edge hilbert_bound_edge

-- P4: D‑RSP extraction
theorem drsp_edge : ∃ alg, polynomial_time alg ∧
    (∀ σ, satisfies Φ σ ↔ alg returns σ) :=
  drsp_correct H_edge
